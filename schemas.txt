CREATE TABLE MESSAGES(
ID SERIAL NOT NULL,
UID TEXT NOT NULL,
TEXT TEXT,
AUTHOR TEXT,
THREAD_ID TEXT,
IS_MEDIA BOOLEAN,
ATTACHMENT_ID TEXT,
ATTACHMENT_LINK TEXT,
TIMESTAMP BIGINT,
PRIMARY KEY(ID));

CREATE TABLE USERS(
UID TEXT NOT NULL,
NAME TEXT NOT NULL,
PROFILEPIC TEXT,
PRIMARY KEY(UID));

CREATE TABLE CaG(
ID SERIAL NOT NULL,
UID TEXT NOT NULL,
THREAD_ID TEXT,
TIMESTAMP BIGINT,
GOING BOOLEAN,
PRIMARY KEY(ID));
